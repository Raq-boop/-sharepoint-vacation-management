### **2. Servi√ßo Microsoft Graph - Integra√ß√£o Avan√ßada**

```typescript
/**
 * GRAPHPHOTOSERVICE - INTEGRA√á√ÉO MICROSOFT GRAPH
 * 
 * Responsabilidades:
 * ‚úÖ Autentica√ß√£o autom√°tica via SPFx
 * ‚úÖ Cache em mem√≥ria para performance  
 * ‚úÖ Batch requests para m√∫ltiplos usu√°rios
 * ‚úÖ Tratamento de erros e fallbacks
 * ‚úÖ Gerenciamento de mem√≥ria (cleanup URLs)
 */

export class GraphPhotoService {
  private _context: WebPartContext;           // üîå Contexto SPFx
  private _graphClient: MSGraphClientV3;      // üåê Cliente Graph v3
  private _photoCache: Map<string, IUserPhoto>; // üíæ Cache fotos

  constructor(context: WebPartContext) {
    this._context = context;
    this._photoCache = new Map();         // üèóÔ∏è Inicializa cache vazio
  }

  /**
   * M√âTODO PRINCIPAL - Busca foto com cache inteligente
   */
  public async getUserPhoto(email: string, userId?: string): Promise<IUserPhoto> {
    const cacheKey = email.toLowerCase();   // üîë Chave normalizada

    // üíæ CACHE HIT - Retorna imediatamente se j√° tem
    if (this._photoCache.has(cacheKey)) {
      console.log('‚úÖ Cache hit para:', email);
      return this._photoCache.get(cacheKey)!;
    }

    // üèóÔ∏è Cria objeto base
    const userPhoto: IUserPhoto = {
      userId: userId || email,
      email: email,
      hasPhoto: false
    };

    try {
      // üåê Obt√©m cliente Graph autenticado
      const graphClient = await this._getGraphClient();
      
      // üìû Chama API do Graph para buscar foto
      const photoResponse = await graphClient
        .api(`/users/${email}/photo/$value`)
        .get();

      if (photoResponse) {
        // üîÑ Converte resposta bin√°ria para blob URL
        const photoUrl = URL.createObjectURL(photoResponse);
        userPhoto.photoUrl = photoUrl;
        userPhoto.hasPhoto = true;
        
        console.log('üì∏ Foto carregada via Graph para:', email);
      }

    } catch (error) {
      console.warn('‚ö†Ô∏è Erro ao buscar foto via Graph:', error);
      
      // üé≠ Gera avatar com iniciais como fallback
      userPhoto.photoUrl = this.getDefaultAvatar(email);
      userPhoto.hasPhoto = false;
    }

    // üíæ Armazena no cache
    this._photoCache.set(cacheKey, userPhoto);
    return userPhoto;
  }

  /**
   * BATCH LOADING - Carrega m√∫ltiplas fotos de uma vez
   */
  public async getUserPhotos(emails: string[]): Promise<Map<string, IUserPhoto>> {
    const results = new Map<string, IUserPhoto>();
    
    // üöÄ Executa todas as buscas em paralelo
    const promises = emails.map(email => this.getUserPhoto(email));
    const photos = await Promise.allSettled(promises);
    
    // üìä Processa resultados
    photos.forEach((result, index) => {
      if (result.status === 'fulfilled') {
        results.set(emails[index], result.value);
      }
    });
    
    return results;
  }

  /**
   * CACHE MANAGEMENT - Limpa cache e libera mem√≥ria
   */
  public clearCache(): void {
    // üßπ Revoga URLs blob para liberar mem√≥ria
    this._photoCache.forEach(photo => {
      if (photo.photoUrl && photo.photoUrl.indexOf('blob:') === 0) {
        URL.revokeObjectURL(photo.photoUrl);
      }
    });
    
    this._photoCache.clear();
    console.log('üßπ Cache de fotos limpo');
  }
}
```
